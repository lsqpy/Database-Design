#资讯表

```
CREATE TABLE `nanhai_article` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `title` varchar(255) NOT NULL COMMENT '标题',
  `short_title` varchar(200) DEFAULT '' COMMENT '短标题',
  `litpic` varchar(50) DEFAULT '' COMMENT '缩略图',
  `seo_title_rule` varchar(50) DEFAULT '' COMMENT '1:关键字+标题+网站名称',
  `keywords` varchar(100) DEFAULT '' COMMENT '关键字',
  `description` varchar(255) DEFAULT '' COMMENT '描述',
  `cid` smallint(6) NOT NULL DEFAULT '0' COMMENT '栏目ID',
  `source` varchar(30) DEFAULT '' COMMENT '来源',
  `redirecturl` varchar(255) DEFAULT '' COMMENT 'URL跳转',
  `content` text COMMENT '内容',
  `sort` int(11) NOT NULL DEFAULT '100' COMMENT '排序',
  `hits` int(11) NOT NULL DEFAULT '0' COMMENT '点击率',
  `status` tinyint(1) NOT NULL DEFAULT '0' COMMENT '点击数',
  `uptime` int(11) NOT NULL DEFAULT '0' COMMENT '更新时间',
  `adtime` int(11) NOT NULL DEFAULT '0' COMMENT '添加时间',
  PRIMARY KEY (`id`),
  KEY `cid` (`cid`),
  KEY `sort` (`sort`),
  KEY `hits` (`hits`),
  KEY `uptime` (`uptime`),
  KEY `addtime` (`adtime`),
  KEY `status` (`status`)
) ENGINE=MyISAM AUTO_INCREMENT=365 DEFAULT CHARSET=utf8;
